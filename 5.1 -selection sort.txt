# Selection Sort using Greedy approach
# The algorithm repeatedly selects the smallest element and puts it in the correct place.

# Take input array from user
arr = list(map(int, input("Enter elements separated by space: ").split()))

n = len(arr)

for i in range(n):
    min_idx = i
    for j in range(i + 1, n):
        # Find the smallest element in remaining unsorted array
        if arr[j] < arr[min_idx]:
            min_idx = j
    # Swap the found minimum with the first element
    arr[i], arr[min_idx] = arr[min_idx], arr[i]

print("Sorted array:", arr)
