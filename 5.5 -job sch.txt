# Job Scheduling Problem using Greedy approach
# Each job has a deadline and profit. We aim to maximize profit within deadlines.

# Take input for number of jobs
n = int(input("Enter number of jobs: "))
jobs = []

for i in range(n):
    name = input(f"Enter job name {i+1}: ")
    deadline = int(input("Enter deadline: "))
    profit = int(input("Enter profit: "))
    jobs.append((name, deadline, profit))

# Sort jobs by profit in descending order
jobs.sort(key=lambda x: x[2], reverse=True)

# Create slots for jobs
max_deadline = max(job[1] for job in jobs)
slots = [False] * max_deadline
result = ['-1'] * max_deadline
total_profit = 0

# Schedule jobs
for job in jobs:
    for j in range(min(max_deadline, job[1]) - 1, -1, -1):
        if not slots[j]:
            slots[j] = True
            result[j] = job[0]
            total_profit += job[2]
            break

print("\nScheduled Jobs:", [j for j in result if j != '-1'])
print("Total Profit:", total_profit)
